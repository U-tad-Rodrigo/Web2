Middleware vs Controller — diferencia clave

Middleware

Funciones que procesan la petición antes del controller. Pueden llamar a next() para pasar al siguiente o next(err) para saltar al errorHandler.

const middleware = (req, res, next) => {
  // procesa algo
  next();         // continúa
  // o
  next(new Error); // va a errorHandler
};

Controller

Función final de la cadena. Habla con la BD y devuelve la respuesta. No llama a next() en el caso feliz.

const controller = async (req, res) => {
  const data = await Model.find({...});
  res.status(200).json({ data });
  // NO llama a next()
};

Express 5 — manejo async nativo

En Express 5 los errors en funciones async se capturan automáticamente y llegan al errorHandler. En Express 4 era necesario un wrapper o try/catch en cada controller.
// Express 5: si el controller lanza, va al errorHandler
export const createClient = async (req, res) => {
  throw AppError.conflict('Ya existe'); // ← Express 5 lo captura solo
};

// Express 4 necesitaría:
// try { ... } catch(err) { next(err) }

export const deleteClient = async (req, res) => {
  const { id } = req.params;
  const soft = req.query.soft === 'true';

export const createDeliveryNoteSchema = z.object({
  client:   z.string(),
  project:  z.string(),

  workDate: z.coerce.date(),     // convierte string "2024-06-01" → Date

Middleware validate() — cómo funciona

// validate.js
export const validate = (schema) => (req, res, next) => {
  const result = schema.safeParse(req.body);
  if (!result.success) {
    // Formatea los errores de Zod como AppError
    return next(AppError.validation('Datos inválidos', result.error.errors));
  }
  req.body = result.data;  // ← reemplaza con valores coercionados y parseados
  next();
};
Clave: req.body = result.data — el body que llega al controller ya está transformado (fechas como Date, campos opcionales limpiados).

Generación de tokens en BildyApp

import jwt from 'jsonwebtoken';
import crypto from 'node:crypto';

// Access token — corta vida, no guardado en BD
const generateAccessToken = (user) =>
  jwt.sign(
    { sub: user._id, role: user.role, company: user.company },
    process.env.JWT_SECRET,
    { expiresIn: process.env.JWT_EXPIRES_IN || '15m' }
  );

// Refresh token — larga vida, guardado HASHEADO en BD
const generateRefreshToken = () => crypto.randomBytes(40).toString('hex');

// Hash SHA256 antes de guardar en BD
const hashToken = (token) =>
  crypto.createHash('sha256').update(token).digest('hex');

// En BD se guarda: user.refreshToken = hashToken(rawToken)
// Al verificar: buscar por hashToken(tokenRecibido)

bcrypt — por qué rounds=12

import bcrypt from 'bcryptjs';

// Al registrar: hashear la contraseña
const hashedPassword = await bcrypt.hash(password, 12);
// rounds=12 → ~250ms en hardware moderno (balance seguridad/velocidad)

// Al login: comparar
const isValid = await bcrypt.compare(plainPassword, user.password);
// bcrypt es LENTO a propósito → dificulta ataques de fuerza bruta
bcrypt incluye salt automático → mismo password da hashes distintos → previene ataques rainbow table.

authenticate() — middleware de JWT

// auth.middleware.js
export const authenticate = async (req, res, next) => {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) throw AppError.unauthorized();

  const token = header.split(' ')[1];
  const payload = jwt.verify(token, process.env.JWT_SECRET); // lanza si inválido

  const user = await User.findById(payload.sub);
  if (!user || user.deleted) throw AppError.unauthorized();

  req.user = user;   // disponible en todos los handlers siguientes
  next();
};

authorize() — middleware de rol

// role.middleware.js
export const authorize = (...roles) => (req, res, next) => {
  if (!roles.includes(req.user.role))
    return next(AppError.forbidden('No tienes permiso'));
  next();
};

Sharp + Cloudinary — cloudinary.service.js

export const uploadImage = async (localPath, folder = 'bildyapp') => {
  if (!isConfigured()) return null; // fallback local

  // Sharp: optimización antes de subir
  const optimized = await sharp(localPath)
    .resize({ width: 800, withoutEnlargement: true }) // máx 800px, no amplía
    .webp({ quality: 80 })                            // convierte a WebP
    .toBuffer();                                       // en memoria

  return new Promise((resolve, reject) => {
    const stream = cloudinary.uploader.upload_stream(
      { folder, resource_type: 'image', format: 'webp' },
      (err, result) => {
        if (err) return reject(err);
        fs.unlink(localPath).catch(() => {}); // borra el archivo temporal
        resolve(result.secure_url);           // devuelve URL pública HTTPS
      }
    );
    stream.end(optimized); // envía el buffer a Cloudinary
  });
};

Pipeline más complejo: horas por proyecto

DeliveryNote.aggregate([
  // 1. Filtra solo albaranes 'hours' de la compañía
  { $match: { company: companyId, deleted: false, format: 'hours' } },

  // 2. Calcula totalHours = hours + suma(workers.hours) en cada documento
  { $project: {
      project: 1,
      totalHours: {
        $add: [
          { $ifNull: ['$hours', 0] },
          { $sum: { $ifNull: ['$workers.hours', []] } }
        ]
      }
  } },

  // 3. Agrupa por proyecto sumando totalHours
  { $group: { _id: '$project', totalHours: { $sum: '$totalHours' } } },

  // 4. JOIN con la colección Projects para traer name + projectCode
  { $lookup: { from: 'projects', localField: '_id', foreignField: '_id', as: 'project' } },
  { $unwind: '$project' },

  // 5. Forma final del documento
  { $project: { _id: 0, projectId: '$_id',
                name: '$project.name', projectCode: '$project.projectCode',
                totalHours: 1 } },
  { $sort: { totalHours: -1 } }
])

export const authenticate = async (req, res, next) => {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) throw AppError.unauthorized();
  const token = header.split(' ')[1];
  const payload = jwt.verify(token, process.env.JWT_SECRET);
  const user = await User.findById(payload.sub);
  if (!user || user.deleted) throw AppError.unauthorized();
  req.user = user;
  next();
};

🔒 Seguridad

Medidas implementadas en BildyApp para proteger la API.

🔑 bcrypt rounds=12

Hash irreversible con salt. ~250ms/hash → dificulta fuerza bruta.

🎫 JWT 15 minutos

Acceso corto. Si robado, expira pronto. Firmado con JWT_SECRET.

🔄 Refresh SHA256

Guardado hasheado en BD. Si se filtra la BD, los tokens no sirven.

🚦 Rate Limit

100 peticiones/15min por IP. Desactivado en NODE_ENV=test.

🧹 NoSQL Sanitize

Elimina $ y . del body. Previene inyección {"$gt":""}.

🪖 Helmet

X-Content-Type-Options, X-Frame-Options, CSP, HSTS...

📁 select: false

Datos no eliminados. Flag deleted: true. Auditable y recuperable.

Diferencias clave entre uploadImage y uploadBuffer:
• uploadImage: resource_type: 'image' + Sharp WebP — para logos y firmas.

• public_id fijado a albaran-<id> hace la subida idempotente: re-firmar sobrescribe el mismo recurso.

📄 PDF Service

Archivo: src/services/pdf.service.js — generación de PDFs de albaranes con pdfkit, dual-mode (stream HTTP + Buffer en memoria).

El servicio expone DOS funciones que comparten un mismo renderDeliveryNote(doc, dn): una hace stream al res (descarga inmediata), otra devuelve un Buffer (para subir a Cloudinary tras firmar).

Por qué dos funciones

🌊 streamDeliveryNotePdf(dn, res)

Pipea el PDF directamente a la respuesta HTTP. No guarda nada. Lo usa GET /pdf/:id cuando el albarán aún no está firmado o no hay pdfUrl.

🧊 buildDeliveryNotePdfBuffer(dn)

Acumula los chunks en memoria y resuelve un Buffer al cerrar el documento. Lo usa el controller tras firmar, para pasarlo a uploadBuffer().

export const buildDeliveryNotePdfBuffer = (dn) =>
  new Promise((resolve, reject) => {
    const doc = new PDFDocument({ margin: 50 });
    const chunks = [];

    doc.on('data',  c => chunks.push(c));
    doc.on('end',   () => resolve(Buffer.concat(chunks)));
    doc.on('error', reject);

    renderDeliveryNote(doc, dn);
    doc.end();
  });

  const url = process.env.SLACK_WEBHOOK_URL;

🚀 index.js — Punto de entrada

El archivo que arranca el servidor: conecta MongoDB, inicia HTTP, gestiona señales del SO.

index.js hace 4 cosas: 1) dbConnect() 2) createHttpServer() + listen() 3) initSocket() 4) SIGTERM/SIGINT graceful shutdown.

src/index.js — código real

import { createHttpServer } from './app.js';
import mongoose from 'mongoose';
import dbConnect from './config/db.js';
import { initSocket, getSocket } from './services/socket.service.js';

const PORT = process.env.PORT || 3000;

const startServer = async () => {
  await dbConnect();              // 1. conecta MongoDB (process.exit(1) si falla)

  const server = createHttpServer(); // 2. crea http.Server desde app Express
  initSocket(server);               // 3. adjunta Socket.io al mismo servidor

  server.listen(PORT, () => {
    console.log(`[server] puerto ${PORT}`);
  });

  const shutdown = async (signal) => {
    console.log(`[shutdown] ${signal}`);
    getSocket()?.close();            // cierra Socket.io primero
    server.close(async () => {       // cierra HTTP (espera conexiones activas)
      await mongoose.disconnect();   // desconecta MongoDB limpiamente
      process.exit(0);
    });
  };

  process.on('SIGTERM', () => shutdown('SIGTERM')); // Docker stop
  process.on('SIGINT',  () => shutdown('SIGINT'));  // Ctrl+C
};

Clase propia que extiende Error para crear errores HTTP estructurados con código y status.

export class AppError extends Error {
  constructor(message, statusCode = 500, code = null) {
    super(message);                    // Error base recibe el mensaje
    this.name = 'AppError';
    this.statusCode = statusCode;      // HTTP status: 400, 401, 404...
    this.code = code;                  // string: 'NOT_FOUND', 'DUPLICATE_KEY'...
    Error.captureStackTrace(this, this.constructor); // excluye este constructor del stack
  }

  // ── Métodos estáticos (factory methods) ──────────────────────────
  static badRequest(msg, code = 'BAD_REQUEST') {
    return new AppError(msg, 400, code);
  }
  static unauthorized(msg = 'No autorizado', code = 'NOT_AUTHORIZED') {
    return new AppError(msg, 401, code);
  }
  static forbidden(msg = 'Acceso denegado', code = 'NOT_ALLOWED') {
    return new AppError(msg, 403, code);
  }
  static notFound(msg = 'Recurso no encontrado', code = 'NOT_FOUND') {
    return new AppError(msg, 404, code);
  }
  static conflict(msg, code = 'CONFLICT') {
    return new AppError(msg, 409, code);
  }
  static validation(msg = 'Error de validación', details = []) {

    return new AppError(msg, 429, code);
  }
  static internal(msg = 'Error interno del servidor', code = 'INTERNAL_ERROR') {
    return new AppError(msg, 500, code);
  }

🛡 Error Handler Centralizado

El middleware de 4 parámetros que captura TODOS los errores de la app y los convierte a respuestas JSON.

El errorHandler tiene exactamente 4 parámetros (err, req, res, next). Express lo reconoce como middleware de error por la aridad (4 params). Si le quitas uno, Express lo trata como middleware normal.

src/middleware/error-handler.js — código real

import mongoose from 'mongoose';
import { AppError } from '../utils/AppError.js';
import { slackError } from '../services/slack.service.js';

// ── 404 (ruta no encontrada) ──────────────────────────────────────────
export const notFound = (req, res) => {
  res.status(404).json({
    error: true, code: 'NOT_FOUND',
    message: `Ruta no encontrada: ${req.method} ${req.originalUrl}`
  });
};

// ── Middleware centralizado (4 params obligatorios en Express) ────────
// eslint-disable-next-line no-unused-vars
export const errorHandler = (err, req, res, _next) => {
  console.error(`[ERROR] [${err.name}] ${err.message}`);

  // 1. AppError (errores controlados del negocio)
  if (err instanceof AppError) {
    const body = { error: true, code: err.code, message: err.message };
    if (err.details) body.details = err.details;
    return res.status(err.statusCode).json(body);
  }

  // 2. Validación Mongoose (campos requeridos, etc.)
  if (err instanceof mongoose.Error.ValidationError) {
    const details = Object.values(err.errors).map(e => e.message);
    return res.status(400).json({ error: true, code: 'VALIDATION_ERROR',
      message: 'Error de validación', details });
  }

  // 3. CastError — ObjectId inválido
  if (err instanceof mongoose.Error.CastError) {
    return res.status(400).json({ error: true, code: 'INVALID_ID',
      message: `ID inválido en '${err.path}'` });
  }

  // 4. Clave duplicada MongoDB (unique index)
  if (err.code === 11000) {
    const field = Object.keys(err.keyValue || {})[0] || 'campo';
    return res.status(409).json({ error: true, code: 'DUPLICATE_KEY',
      message: `Ya existe un registro con ese '${field}'` });
  }

  // 5. Multer: archivo demasiado grande
  if (err.code === 'LIMIT_FILE_SIZE') {
    return res.status(400).json({ error: true, code: 'FILE_TOO_LARGE',
      message: 'El archivo supera el límite de 5 MB' });
  }

  // 6. Multer: tipo no permitido
  if (err.message === 'INVALID_FILE_TYPE') {
    return res.status(415).json({ error: true, code: 'INVALID_FILE_TYPE',
      message: 'Tipo de archivo no permitido' });
  }

  // 7. JSON malformado
  if (err instanceof SyntaxError && err.status === 400 && 'body' in err) {
    return res.status(400).json({ error: true, code: 'INVALID_JSON',
      message: 'JSON inválido en el cuerpo' });
  }

  // 8. Error genérico (5XX → alerta Slack)
  const status = err.status || err.statusCode || 500;
  if (status >= 500) {
    slackError({ method: req.method, path: req.originalUrl,
      status, message: err.message, stack: err.stack }).catch(() => {});
  }
  res.status(status).json({
    error: true, code: 'INTERNAL_ERROR',
    message: process.env.NODE_ENV === 'production'
      ? 'Error interno del servidor' : err.message
  });
};

¿Qué es EventEmitter?

Clase nativa de Node.js (node:events) que implementa el patrón pub/sub (publish/subscribe). Un emisor lanza eventos con datos; múltiples listeners reaccionan. Todo síncrono en el mismo proceso (distinto a Socket.io que es red).

import { EventEmitter } from 'node:events';

const emitter = new EventEmitter();

// Suscripción (listener)
emitter.on('user:registered', (data) => {
  console.log('Nuevo usuario:', data.email);
});

// Emisión (disparo del evento)
emitter.emit('user:registered', { email: 'user@example.com' });
// → Nuevo usuario: user@example.com

Métodos clave

MétodoQué hace
.on(event, fn)Registra listener (permanente)
.once(event, fn)Listener que se elimina tras primera llamada
.emit(event, ...args)Dispara el evento. Llama a todos los listeners síncronamente.
.off(event, fn)Elimina un listener específico
.removeAllListeners(event)Elimina todos los listeners de un evento
.listenerCount(event)Número de listeners para un evento
Por defecto, un EventEmitter permite máximo 10 listeners por evento antes de mostrar warning de memory leak. Usa emitter.setMaxListeners(n) para cambiar el límite.

¿Qué es un Stream?

Abstracción de Node.js para procesar datos en trozos (chunks) en vez de cargar todo en memoria. Implementa EventEmitter internamente — emite eventos data, end, error.

Tipos de Streams

Readable

Solo lectura. Ejemplos: fs.createReadStream(), req HTTP, respuesta de fetch.

Writable

Solo escritura. Ejemplos: fs.createWriteStream(), res HTTP, process.stdout.

Transform

Lee y transforma. Ejemplos: zlib.createGzip(), Sharp transforms.

Duplex

Bidireccional. Ejemplos: WebSocket, TCP socket.

pipe() — conectar streams

// Sin stream (todo en memoria — MAL para archivos grandes):
const content = fs.readFileSync('file.txt');   // carga TODO en RAM
res.send(content);

// Con stream (chunk a chunk — BIEN):
fs.createReadStream('file.txt').pipe(res);    // stream → response

// Encadenamiento de streams con pipe():
fs.createReadStream('input.txt')
  .pipe(zlib.createGzip())     // Transform: comprime
  .pipe(res);                  // Writable: envía al cliente

🔴 Errores críticos de concepto

El JWT payload usa payload.id, NO payload.sub. El token se firma con { id: user._id } → al verificarlo tienes payload.id. Si pones payload.sub, el código falla.

// ✅ CORRECTO en BildyApp
const user = await User.findById(payload.id);

// ❌ INCORRECTO

if (!result.success) return next(AppError.validation(..., result.error.errors));
req.body = result.data;

try {
  req.body = schema.parse(req.body);  // lanza ZodError si falla

  if (err instanceof ZodError) return next(AppError.validation(..., details));
  next(err);

🟠 Trampas de orden y flujo

El graceful shutdown cierra en orden: Socket.io → HTTP server → MongoDB. Si cierras MongoDB primero, las peticiones en vuelo fallan con "connection closed". Si cierras HTTP antes que Socket.io, quedan conexiones WebSocket colgadas.
El middleware en Express se ejecuta en orden de registro. Si pones errorHandler antes de las rutas, los errores no llegan a él. El orden correcto: seguridad → rutas → notFound → errorHandler.
// ✅ Orden correcto en app.js
app.use(helmet());          // 1. Seguridad
app.use(cors());            // 2. CORS
app.use(express.json());    // 3. Parser
app.use('/api', routes);    // 4. Rutas
app.use(notFound);          // 5. 404 (después de rutas)
app.use(errorHandler);      // 6. Errores (siempre último)
En el Dockerfile multi-stage, el CMD no usa npm start — usa node --env-file=.env src/index.js directamente. Esto evita la sobrecarga de npm y aprovecha el flag nativo de Node 20 para cargar .env.
